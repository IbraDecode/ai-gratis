Contoh cURL

curl -X POST ${SIbraAI_URL || 'http://localhost:3000'}/api/chat \

curl -N -X POST ${SIbraAI_URL || 'http://localhost:3000'}/api/chat \
  -H "Content-Type: application/json" \
  -d '{"prompt":"cerita pendek","model":"gemini","stream":true}'
